4 // Sound Magic Definition:
5 // First Byte (Upper Nibble): Sound Bank
6 // First Byte (Lower Nibble): Bitflags for audio playback?
7 // Second Byte: Sound ID
8 // Third Byte: Bitflags for audio properties
9 // Fourth Byte (Upper Nibble): Envelope? (0-8, 8 is play the full sound)
10 // Fourth Byte (Lower Nibble): Sound Status (this is set to SOUND_PLAYING when passed to the audio driver.)
11 #define SOUND_ARG_LOAD(bank, playFlags, soundID, propFlags, envelope) (((u32) (bank) << 28) | \
12  ((u32) (playFlags) << 24) | ((u32) (soundID) << 16) | ((u32) (propFlags) << 8) | \
13  ((u32) (envelope) << 4) | SOUND_STARTING)
